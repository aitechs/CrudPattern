using System;
using Dapper.Contrib.Extensions;
using Aitech.CrudPattern;

namespace CrudPattern
{
    [Table("Students_History")]
    public class StudentHistory: Entity
    {		
		StudentHistory Original;
	#region Default Properties
		        
		private int _studentid;        
        public int StudentId 
		{ 
			get { return  _studentid; } 
			set 
			{ 
				if ( _studentid != value) {  _studentid = value; OnChanged(value); }  
			}
		}

        
		private string _school;        
        public string School 
		{ 
			get { return  _school; } 
			set 
			{ 
				if ( _school != value) {  _school = value; OnChanged(value); }  
			}
		}

        
		private string _batch;        
        public string Batch 
		{ 
			get { return  _batch; } 
			set 
			{ 
				if ( _batch != value) {  _batch = value; OnChanged(value); }  
			}
		}

        
		private string _semester;        
        public string Semester 
		{ 
			get { return  _semester; } 
			set 
			{ 
				if ( _semester != value) {  _semester = value; OnChanged(value); }  
			}
		}

        
		private string _coursecode;        
        public string CourseCode 
		{ 
			get { return  _coursecode; } 
			set 
			{ 
				if ( _coursecode != value) {  _coursecode = value; OnChanged(value); }  
			}
		}

        
		private int _yearlevel;        
        public int YearLevel 
		{ 
			get { return  _yearlevel; } 
			set 
			{ 
				if ( _yearlevel != value) {  _yearlevel = value; OnChanged(value); }  
			}
		}

        
		private string _sectionname;        
        public string SectionName 
		{ 
			get { return  _sectionname; } 
			set 
			{ 
				if ( _sectionname != value) {  _sectionname = value; OnChanged(value); }  
			}
		}

	#endregion
	
		protected void BeginTrackingChanges()
		{
			Original = new StudentHistory ();
			        
			Original.StudentId = this.StudentId;        
			Original.School = this.School;        
			Original.Batch = this.Batch;        
			Original.Semester = this.Semester;        
			Original.CourseCode = this.CourseCode;        
			Original.YearLevel = this.YearLevel;        
			Original.SectionName = this.SectionName;		}

		protected void CancelChanges()
		{
			        
			this.StudentId = Original.StudentId;        
			this.School = Original.School;        
			this.Batch = Original.Batch;        
			this.Semester = Original.Semester;        
			this.CourseCode = Original.CourseCode;        
			this.YearLevel = Original.YearLevel;        
			this.SectionName = Original.SectionName;		}
	}


	
}
